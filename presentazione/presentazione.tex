\begin{frame}{1. Obiettivi, Dataset e Preprocessing}
  \begin{columns}
    \begin{column}{0.53\textwidth}
      \textbf{Il Dataset \& Obiettivi:}
      \begin{itemize}
        \item \textbf{Dataset}: Telco Customer Churn (IBM, 7043 clienti, classi sbilanciate 3:1).
        \item \textbf{Obiettivo}: Prevedere l'abbandono dei clienti (\textit{Churn Prediction}) per fidelizzarli.
        \item \textbf{Riproducibilità}: Codice interamente descritto ed eseguibile in \texttt{segmentazione\_crunch.ipynb}.
      \end{itemize}
      \vspace{0.15cm}
      \textbf{Preprocessing dei Dati:}
      \begin{itemize}
        \item Imputazione di $11$ righe vuote in \textit{TotalCharges} con \texttt{0.0} (nuovi clienti con tenure=0).
        \item Rimozione del codice identificativo univoco \textit{customerID}.
        \item Codifica categoriche (OHE con \texttt{drop\_first=True}).
        \item Divisione Train/Test (80/20) stratificata prima dello scaling.
        \item Standardizzazione delle feature numeriche.
      \end{itemize}
    \end{column}

      \begin{center}\scriptsize \textbf{Grafico}: Chi abbandona (Yes Churn) ha una permanenza bassissima (mediana $\approx 10$ mesi).\end{center}

  \end{columns}
\end{frame}

% Slide 4: Modelli e Scenari di Classificazione
\begin{frame}{3. Modelli di Classificazione e Scenari}
  \textbf{Tre classificatori con approcci differenti:}
  \vspace{0.15cm}
  \begin{columns}[t]
    \begin{column}{0.31\textwidth}
      \begin{block}{1. Regressione Logistica}
        \scriptsize
        \textbf{Baseline lineare:}\\
        Mappa il punteggio lineare delle feature in una probabilità tramite la curva \textbf{Sigmoide}.
      \end{block}
    \end{column}
    \begin{column}{0.31\textwidth}
      \begin{block}{2. Albero di Decisione}
        \scriptsize
        \textbf{White-Box Model:}\\
        Crea regole IF-THEN gerarchiche basate sull'indice di Gini per massimizzare la purità.
      \end{block}
    \end{column}
    \begin{column}{0.31\textwidth}
      \begin{block}{3. Rete Neurale (MLP)}
        \scriptsize
        \textbf{Black-Box Model:}\\
        Usa strati nascosti con attivazione non lineare (ReLU) ed ottimizzatore Adam per pattern complessi.
      \end{block}
    \end{column}
  \end{columns}
  
  \vspace{0.3cm}
  \textbf{I due scenari sperimentali messi a confronto:}
  \vspace{0.1cm}
  \begin{itemize}
    \item \textbf{Scenario A (Senza Cluster)}: Utilizzo delle sole feature originali preelaborate.
    \item \textbf{Scenario B (Con Cluster)}: Aggiunta del cluster di appartenenza ($0$ o $1$) come feature per guidare la classificazione.
  \end{itemize}
\end{frame}

% Slide 5: Funzionamento Concettuale dei Modelli
\begin{frame}{3.1 Come Funzionano i Modelli? (Concetti Chiave)}
  \textbf{Ogni classificatore elabora i dati con una logica differente:}
  \vspace{0.2cm}
  
  \begin{columns}[t]
    \begin{column}{0.32\textwidth}
      \begin{block}{Regressione Logistica}
        \scriptsize
        \textbf{Logica Probabilistica:}\\
        \begin{itemize}
          \item Pesa l'influenza di ciascuna variabile sul rischio di abbandono.
          \item Mappa le variabili in una probabilità complessiva.
          \item Classifica come "a rischio" se supera la soglia del 50\%.
        \end{itemize}
      \end{block}
    \end{column}
    
    \begin{column}{0.32\textwidth}
      \begin{block}{Albero di Decisione}
        \scriptsize
        \textbf{Logica a Regole (IF-THEN):}\\
        \begin{itemize}
          \item Divide ricorsivamente i dati ponendo domande basate su soglie.
          \item Cerca lo split che crea i gruppi più omogenei.
          \item Struttura visiva trasparente e intuitiva per le decisioni.
        \end{itemize}
      \end{block}
    \end{column}
    
    \begin{column}{0.32\textwidth}
      \begin{block}{Rete Neurale (MLP)}
        \scriptsize
        \textbf{Logica Multi-Strato:}\\
        \begin{itemize}
          \item Simula una rete di neuroni organizzata in vari livelli.
          \item Apprende relazioni non lineari complesse con la ReLU.
          \item Regola continuamente le connessioni per correggere gli errori.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% Slide 7: Interpretazione dell'Albero di Decisione
\begin{frame}{5. Interpretazione dell'Albero di Decisione}
  \footnotesize
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{Significato dei Colori:}
      \begin{itemize}
        \item \textbf{Nodi Blu}: Maggioranza di clienti fedeli (No Churn).
        \item \textbf{Nodi Arancioni}: Maggioranza di clienti a rischio (Yes Churn).
        \item \textbf{Intensità}: Colori più scuri indicano decisioni più nette e omogenee.
      \end{itemize}
      \vspace{0.15cm}
      
      \textbf{Le Regole Principali:}
      \begin{itemize}
        \item \textbf{1. Contratto Mensile}: Lo split fondamentale. Chi ha un contratto mensile è molto più instabile.
        \item \textbf{2. Tenure (Permanenza)}: I clienti recenti (tenure bassa) con contratto mensile sono a rischio critico.
        \item \textbf{3. Servizio Internet}: L'uso di Fibra Ottica è legato a spesa alta e maggior abbandono.
      \end{itemize}
    \end{column}

% Slide 8: Diagnostica dei Modelli e Importanza Feature
\begin{frame}{6. Diagnostica dei Modelli e Importanza Feature}
  \footnotesize
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{Comportamento e Diagnostica:}
      \begin{itemize}
        \item \textbf{Underfitting}: Escluso. I modelli lineari e l'albero catturano bene i pattern base ($\approx 80\%$ accuratezza).
        \item \textbf{Overfitting}: Rilevato nella Rete Neurale (MLP). Raggiunge il $91\%$ sul train ma cala al $73.6\%$ sul test, imparando a memoria i dati.
      \end{itemize}
      \vspace{0.15cm}
      
      \textbf{L'utilità del Cluster:}
      \begin{itemize}
        \item Anche in presenza di overfitting, la feature di clustering funge da forte segnale d'aiuto per l'MLP, guidando l'apprendimento verso i veri clienti a rischio e aumentando la Recall di test di oltre $15$ punti.
      \end{itemize}
    \end{column}

      \centering
      \includegraphics[width=\textwidth]{../grafici/feature_importances.png}

% Slide 9: Conclusioni e Sviluppi Futuri
\begin{frame}{7. Conclusioni e Sviluppi Futuri}
  \footnotesize
  \textbf{Lezioni principali:}
  \begin{itemize}
    \item \textbf{Clustering k-Means}: Identifica due profili commerciali chiari e coerenti.
    \item \textbf{Valore Modello Ibrido}: L'inserimento del cluster aumenta drasticamente la sensibilità (Recall $+15.5\%$) dell'MLP nell'intercettare l'abbandono.
    \item \textbf{Trasparenza vs Performance}: L'albero di decisione è white-box ed immediatamente interpretabile; la rete neurale MLP è black-box e soffre di overfitting ma offre margini di ottimizzazione migliori.
  \end{itemize}
  \vspace{0.15cm}
  
  \textbf{Sviluppi Futuri (Miglioramento dei Modelli):}
  \begin{itemize}
    \item \textbf{Regolarizzazione dell'MLP}: Introdurre penalizzazione L2 o dropout per contrastare l'overfitting.
    \item \textbf{Bilanciamento}: Applicare tecniche come SMOTE per compensare lo sbilanciamento del dataset (3:1).
  \end{itemize}
\end{frame}
